##### HOW TO RUN ".\add_quality_data_partial_script.py" ON CLOUD #####

# APT-GET UPDATE, INSTALL SCREEN.
sudo apt-get update
sudo apt-get install screen

# DOWNLOAD REPOSITORY
sudo apt-get install git
git clone https://github.com/ansellim/6242.git

# INSTALL CONDA
sudo apt install conda
source /opt/conda/etc/profile.d/conda.sh

# CONFIGURE CONDA
source /opt/conda/etc/profile.d/conda.sh
conda create -n myenv numpy pandas requests
conda activate myenv
pip install python-dotenv geopy sklearn

# ADD .ENV FILE MANUALLY, AND ADD GOOGLE_API_KEY
nano ../.env

# RUN SCRIPT
screen
source /opt/conda/etc/profile.d/conda.sh
conda activate myenv
python add_quality_data_partial_script.py

# TO EXIT, JUST CTRL + A + D. TO RE-ENTER SCREEN, type "control -r"
